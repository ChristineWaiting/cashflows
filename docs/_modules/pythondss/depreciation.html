[docs]def print_depr(depr, adepr, begbook, endbook):
    """Prints a depreciation table

    Args:
       cost (int, float): Initial cost of the asset
       depr (list): Depreciation per period
       adepr (list): Accumulated depreciation per period
       begbook (list): Beginning book value
       endbook (list): Ending book value

    Returns:
       None

    """
    txt = [' t      Beginning  Depreciation   Accumulated        Ending',
           '       Book Value                Depreciation    Book Value',
           '------------------------------------------------------------']

    for time, _ in enumerate(depr):

        fmt = ' {:<2d} {:13.2f} {:13.2f} {:13.2f} {:13.2f}'
        txt.append(fmt.format(time,
                              begbook[time],
                              depr[time],
                              adepr[time],
                              endbook[time]))

    print('\n'.join(txt))




[docs]def depreciation_sl(cost=None, life=None, salvalue=0, quiet=True):
    """Computes the depreciation of an asset using straight line depreciation
    method.

    Args:
        cost (int, float, list): the cost per period of the assets.
        life (int): number of depreciation periods for the asset.
        salvalue(int, float): salvage value.
        quiet (bool): when True, the procedure prints a depreciation table.

    Returns:
        A tuple (dep, accum) of lists (tuple): depreciation per period and accumulated depreciation per period

    """
    if not isinstance(cost, (int, float, list)):
        raise TypeError('Invalid type for `cost`')
    if not isinstance(life, (int, float, list)):
        raise TypeError('Invalid type for `life`')
    if not isinstance(salvalue, (int, float, list)):
        raise TypeError('Invalid type for `salvalue`')


    if isinstance(cost, (int, float)):
        maxlen = 1
        cost = [cost]
    else:
        maxlen = len(cost)

    if isinstance(life, (int, float)):
        life = [life] * maxlen
    elif isinstance(life, list) and len(life) != maxlen:
        raise ValueError('List in parameters must have the same length')

    if isinstance(salvalue, (int, float)):
        salvalue = [salvalue] * maxlen
    elif isinstance(salvalue, list) and len(salvalue) != maxlen:
        raise ValueError('List in parameters must have the same length')

    depr = [0] * (max(life) + maxlen - 1)
    adepr = [0] * (max(life) + maxlen - 1)
    begbook = [0] * (max(life) + maxlen - 1)
    endbook = [0] * (max(life) + maxlen - 1)

    for index in range(maxlen):
        xdepr = [(cost[index] - salvalue[index]) / life[index]] * life[index]
        for time in range(life[index]):
            depr[index + time] += xdepr[time]

    for time, _ in enumerate(depr):
        if time > 0:
            adepr[time] = adepr[time - 1] + depr[time]
        else:
            adepr[time] = depr[time]

    for time, _ in enumerate(depr):
        if time > 0:
            begbook[time] = endbook[time - 1]
        if time < len(cost):
            begbook[time] += cost[time]
        endbook[time] = begbook[time] - depr[time]

    if quiet is True:
        return (depr, adepr)

    print_depr(depr, adepr, begbook, endbook)

    """Computes the depreciation of an asset using the sum-of-year's-digits
    method.

    Args:
        cost (int, float, list): the cost per period of the assets.
        life (int): number of depreciation periods for the asset.
        salvalue(int, float): salvage value.
        quiet (bool): when True, the procedure prints a depreciation table.

    Returns:
        A tuple (dep, accum) of lists (tuple): depreciation per period and accumulated depreciation per period

    """
    if not isinstance(cost, (int, float, list)):
        raise TypeError('Invalid type for `cost`')
    if not isinstance(life, (int, float, list)):
        raise TypeError('Invalid type for `life`')
    if not isinstance(salvalue, (int, float, list)):
        raise TypeError('Invalid type for `salvalue`')

    if isinstance(cost, (int, float)):
        maxlen = 1
        cost = [cost]
    else:
        maxlen = len(cost)

    if isinstance(life, (int, float)):
        life = [life] * maxlen
    elif isinstance(life, list) and len(life) != maxlen:
        raise ValueError('List in parameters must have the same length')

    if isinstance(salvalue, (int, float)):
        salvalue = [salvalue] * maxlen
    elif isinstance(salvalue, list) and len(salvalue) != maxlen:
        raise ValueError('List in parameters must have the same length')

    depr = [0] * (max(life) + maxlen - 1)
    adepr = [0] * (max(life) + maxlen - 1)
    begbook = [0] * (max(life) + maxlen - 1)
    endbook = [0] * (max(life) + maxlen - 1)

    for index in range(maxlen):

[docs]def depreciation_db(cost=None, life=5, salvalue=0, factor=1, convert_to_sl=False, quiet=True):
    """Computes the depreciation of an asset using the declining balance
    method.

    Args:
        cost (int, float, list): the cost per period of the assets.
        life (int): number of depreciation periods for the asset.
        salvalue (int, float): salvage value.
        factor (float): acelerating factor for depreciation.
        convert_to_sl (bool): converts to straight line method?
        quiet (bool): when True, the procedure prints a depreciation table.

    Returns:
        A tuple (dep, accum) of lists (tuple): depreciation per period and accumulated depreciation per period


    """

    if not isinstance(cost, (int, float, list)):
        raise TypeError('Invalid type for `cost`')
    if not isinstance(life, (int, float, list)):
        raise TypeError('Invalid type for `life`')
    if not isinstance(salvalue, (int, float, list)):
        raise TypeError('Invalid type for `salvalue`')
    if not isinstance(factor, (int, float, list)):
        raise TypeError('Invalid type for `factor`')
    if not isinstance(convert_to_sl, bool):
        raise TypeError('Invalid type for `convert_to_sl`')


    if isinstance(cost, (int, float)):
        maxlen = 1
        cost = [cost]
    else:
        maxlen = len(cost)

    if isinstance(life, (int, float)):
        life = [life] * maxlen
    elif isinstance(life, list) and len(life) != maxlen:
        raise ValueError('List in parameters must have the same length')

    if isinstance(salvalue, (int, float)):
        salvalue = [salvalue] * maxlen
    elif isinstance(salvalue, list) and len(salvalue) != maxlen:
        raise ValueError('List in parameters must have the same length')

    if isinstance(factor, (int, float)):
        factor = [factor] * maxlen
    elif isinstance(factor, list) and len(factor) != maxlen:
        raise ValueError('List in parameters must have the same length')

    depr = [0] * (max(life) + maxlen - 1)
    adepr = [0] * (max(life) + maxlen - 1)
    begbook = [0] * (max(life) + maxlen - 1)
    endbook = [0] * (max(life) + maxlen - 1)

    for index in range(maxlen):

        factor[index] = factor[index] / life[index]

        rem_cost = cost[index]
        xdepr = [0] * life[index]

        sl_depr = (cost[index] - salvalue[index]) / life[index]

        for time in range(life[index]):
            xdepr[time] = rem_cost * factor[index]
            if convert_to_sl is True and xdepr[time] < sl_depr:
                xdepr[time] = sl_depr
            rem_cost -= xdepr[time]
            if rem_cost < salvalue[index]:
                rem_cost += xdepr[time]
                xdepr[time] = rem_cost - salvalue[index]
                rem_cost = salvalue[index]

        for time in range(life[index]):
            depr[index + time] += xdepr[time]

    for time, _ in enumerate(depr):
        if time > 0:
            adepr[time] = adepr[time - 1] + depr[time]
        else:
            adepr[time] = depr[time]

    for time, _ in enumerate(depr):
        if time > 0:
            begbook[time] = endbook[time - 1]
        if time < len(cost):
            begbook[time] += cost[time]
        endbook[time] = begbook[time] - depr[time]

    if quiet is True:
        return (depr, adepr)

    print_depr(depr, adepr, begbook, endbook)
