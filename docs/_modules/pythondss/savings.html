[docs]def savings(deposits, start=0, rate=0, nper=1, initbal=0, quiet=True):
    """
    Args:
        deposits (list): deposits to the account.
        start (int): initial period for the deposits
        rate (float, list): interest rate paid by the account.
        nper (int): total number of periods to simulate.
        initbal (float): initial balance of the account.
        quie (bool): prints summary report?

    Return:
        interes, ending_balance (list, list): earned interest and final balance.

    """
    if isinstance(deposits, (int, float)) and nper == 1:
        raise ValueError('deposits must be a list or `nper` must be > 1')

    if not isinstance(nper, int):
        raise ValueError('`nper` must be an integer')

    if not isinstance(rate, (int, float, list)):
        raise TypeError('Invalid type for `rate`')

    if isinstance(deposits, list):
        nper = max(nper, start + len(deposits))

    if isinstance(deposits, (int, float)):
        deposits = [0] * start + [deposits] * (nper - start)
    elif len(deposits) < nper:
        deposits = [0] * start + deposits + [0] * (nper - len(deposits) - start)

    if isinstance(rate, (int, float)):
        rate = [rate] * nper

    if len(rate) < nper:
        rate += [rate[-1]] * (nper - len(rate))
        # raise ValueError('len(rate) < nper')

    begbal = [0] * nper
    interest = [0] * nper
    endbal = [0] * nper


    ##
    ## balance calculation
    ##
    for time in range(nper):

        if time == 0:
            #
            begbal[time] = initbal
            interest[time] = begbal[time] * rate[time] / 100
            endbal[time] = begbal[time] + deposits[time] + interest[time]
            #
        else:
            #
            begbal[time] = endbal[time - 1]
            interest[time] = begbal[time] * rate[time] / 100

            if deposits[time] < 0 and -deposits[time] > begbal[time] + interest[time]:
                #
                deposits[time] = -(begbal[time] + interest[time])
                #

            endbal[time] = begbal[time] + deposits[time] + interest[time]
            #

    if quiet is True:
        return (interest, endbal)

    txt = []
    txt.append(' t     Beginning      Deposit       Earned       Ending')
    txt.append('         balance                  Interest      balance')
    txt.append('------------------------------------------------------------')

    for time, _ in enumerate(deposits):

        fmt = ' {:<3d} {:12.2f} {:12.2f} {:12.2f} {:12.2f}'
        txt.append(fmt.format(time,
                              begbal[time],
                              deposits[time],
                              interest[time],
                              endbal[time]))

    return print('\n'.join(txt))
